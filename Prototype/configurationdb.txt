CREATE DATABASE AllConfig;

USE AllConfig;

CREATE TABLE minor_configs(id INT PRIMARY KEY AUTO_INCREMENT, database_name VARCHAR(255), server_name VARCHAR(255));

CREATE TABLE users(id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255), password_digest VARCHAR(255), satellite_name VARCHAR(255), admin BOOLEAN NOT NULL DEFAULT 0);

CREATE TABLE `versions` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT,
  `item_type` varchar(191) NOT NULL,
  `item_id` int(11) NOT NULL,
  `event` varchar(255) NOT NULL,
  `whodunnit` varchar(255) DEFAULT NULL,
  `object` longtext,
  `created_at` datetime DEFAULT NULL,
  `object_changes` longtext,
  PRIMARY KEY (`id`),
  KEY `index_versions_on_item_type_and_item_id` (`item_type`,`item_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
