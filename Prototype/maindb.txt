CREATE TABLE SUBSYSTEM_TBL(SUBSYSTEM_ID INT PRIMARY KEY, SUBSYSTEM_NAME VARCHAR(255));

CREATE TABLE UI_TYPES_TBL(UI_TYPE_ID INT PRIMARY KEY, UI_TYPE VARCHAR(255));

CREATE TABLE CMD_TYPES_TBL(CMD_TYPE_ID VARCHAR(255) PRIMARY KEY, CMD_TYPE VARCHAR(255), CMD_TYPES_TBL_CMD_TYPE VARCHAR(255));

CREATE TABLE CONFIRMER_TBL(POSITION VARCHAR(255) PRIMARY KEY, NAME VARCHAR(255));

CREATE TABLE BASE_CMD_TBL(CMD_ID INT PRIMARY KEY AUTO_INCREMENT, DESCRIPTION VARCHAR(255), SUBSYSTEM INT, FORMATION_TYPE VARCHAR(255), CMD_TYPE VARCHAR(255), BASE_CMD_CODE VARCHAR(255), BASE_CMD_MASK VARCHAR(255), CRITICALITY VARCHAR(255), PULSE_WIDTH VARCHAR(255), MAP_ID VARCHAR(255), UI_TYPE INT, PREFERRABLE_DECODER VARCHAR(255), CONFIRMATION_TYPE VARCHAR(255), REMARKS VARCHAR(255), FOREIGN KEY(SUBSYSTEM) REFERENCES SUBSYSTEM_TBL(SUBSYSTEM_ID) ON DELETE CASCADE ON UPDATE CASCADE, FOREIGN KEY(UI_TYPE) REFERENCES UI_TYPES_TBL(UI_TYPE_ID) ON DELETE CASCADE ON UPDATE CASCADE, FOREIGN KEY(CMD_TYPE) REFERENCES CMD_TYPES_TBL(CMD_TYPE_ID) ON DELETE CASCADE ON UPDATE CASCADE);

CREATE TABLE EXPRESSION(EXPR_ID INT, EXPR_TYPE VARCHAR(255), TM_PID VARCHAR(255), REL_OP VARCHAR(255), VALUE VARCHAR(255), VALUE_TYPE VARCHAR(255), PRIMARY KEY(EXPR_ID));

CREATE TABLE PREREQ_CONF_EXPR_TBL(CMD_ID INT, ENTRY_TYPE VARCHAR(255), FIELD_NO INT, EXPR_ID INT, LOGICAL_OP VARCHAR(255), PRIMARY KEY(CMD_ID, ENTRY_TYPE, FIELD_NO), FOREIGN KEY(EXPR_ID) REFERENCES EXPRESSION(EXPR_ID) ON DELETE CASCADE ON UPDATE CASCADE, FOREIGN KEY(CMD_ID) REFERENCES BASE_CMD_TBL(CMD_ID) ON DELETE CASCADE ON UPDATE CASCADE);

CREATE TABLE CPID_TBL(CMD_ID INT, CPID VARCHAR(255), CPID_DESC VARCHAR(255), TM_VAL VARCHAR(255), TM_COUNT VARCHAR(255), PROC_TYPE VARCHAR(255), PRIMARY KEY(CMD_ID, CPID), FOREIGN KEY(CMD_ID) REFERENCES BASE_CMD_TBL(CMD_ID));

CREATE TABLE CMD_PART_TYPES_TBL(CMD_PART_TYPE_ID VARCHAR(255) PRIMARY KEY, CMD_PART_TYPE VARCHAR(255));

CREATE TABLE CMD_PART_TBL(CMD_ID INT, CMD_PART_NO VARCHAR(255), CMD_PART_MNEMONIC VARCHAR(255), CMD_PART_TYPE VARCHAR(255), START_BIT VARCHAR(255), NO_OF_BITS VARCHAR(255), PART_MASK VARCHAR(255), RESOLUTION VARCHAR(255), OFFSET VARCHAR(255), INPUT_FORMAT VARCHAR(255), LL VARCHAR(255), UL VARCHAR(255), UNITS VARCHAR(255), REMARKS VARCHAR(255), PRIMARY KEY(CMD_ID, CMD_PART_NO), FOREIGN KEY(CMD_PART_TYPE) REFERENCES CMD_PART_TYPES_TBL(CMD_PART_TYPE_ID) ON DELETE CASCADE ON UPDATE CASCADE ,FOREIGN KEY(CMD_ID) REFERENCES BASE_CMD_TBL(CMD_ID) ON DELETE CASCADE ON UPDATE CASCADE);

CREATE TABLE BIT_SEL_TBL_INFO(BIT_SEL_TBL_NO INT PRIMARY KEY, BIT_SEL_TBL_NAME VARCHAR(255));

CREATE TABLE BIT_SEL_TBL(BIT_SEL_TBL_NO INT, FIELD_NO DOUBLE , START_BIT DOUBLE, NO_OF_BITS VARCHAR(255), BIT_MASK VARCHAR(255), VALUE VARCHAR(255), FIELD_MNEMONICS VARCHAR(255), VALUE_MNEMONICS VARCHAR(255), LOGICAL_OP VARCHAR(255), PRIMARY KEY(BIT_SEL_TBL_NO, FIELD_NO), FOREIGN KEY(BIT_SEL_TBL_NO) REFERENCES BIT_SEL_TBL_INFO(BIT_SEL_TBL_NO) ON DELETE CASCADE ON UPDATE CASCADE);

CREATE TABLE LKUP_TBL_INFO(LKUP_TBL_NO INT PRIMARY KEY, LKUP_TBL_NAME VARCHAR(255), LKUP_TBL_TYPE VARCHAR(255));

CREATE TABLE LKUP_TBL(LKUP_TBL_NO INT, LKUP_ENTRY_INDEX VARCHAR(255), BYTE_CODE VARCHAR(255), VALUE VARCHAR(255), PRIMARY KEY(LKUP_TBL_NO, LKUP_ENTRY_INDEX), FOREIGN KEY(LKUP_TBL_NO) REFERENCES LKUP_TBL_INFO(LKUP_TBL_NO) ON DELETE CASCADE ON UPDATE CASCADE);


CREATE TABLE MAP_ID_TBL(MAP_ID VARCHAR(32) PRIMARY KEY);
CREATE TABLE PART_TYPE_TBL(PART_TYPE VARCHAR(32) PRIMARY KEY);
CREATE TABLE INPUT_FORMAT_TBL(INPUT_FORMAT VARCHAR(32) PRIMARY KEY);
CREATE TABLE PREFERRABLE_DECODER_TBL(PREFERRABLE_DECODER VARCHAR(32) PRIMARY KEY);
CREATE TABLE CRITICALITY_TBL(CRITICALITY VARCHAR(32) PRIMARY KEY);
CREATE TABLE PULSE_WIDTH_TBL(PULSE_WIDTH VARCHAR(32) PRIMARY KEY);
CREATE TABLE BASE_FORMATION_LOGIC_TBL(BASE_FORMATION_LOGIC VARCHAR(32) PRIMARY KEY);
CREATE TABLE LK_TBL_TYPE_TBL(LK_TBL_TYPE VARCHAR(32) PRIMARY KEY);
CREATE TABLE LOGICAL_OP_TBL(LOGICAL_OP VARCHAR(32) PRIMARY KEY);
